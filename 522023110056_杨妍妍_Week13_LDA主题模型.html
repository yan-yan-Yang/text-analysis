    import matplotlib.pyplot as plt
    scores = []
    for k in range(min_k, max_k):
        #seed
        mdl = tp.LDAModel(min_df=min_df, k=k, seed=555)
        for words in docs:

        coh = tp.coherence.Coherence(mdl)

    #print(x)
    #print()
